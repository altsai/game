18    @Test
19    public void testEquals() {
20      Actor a = new Actor("Jessica Biel", "12345");
21      Actor b = new Actor("Jessica Biel", "67890");
22      Actor c = new Actor("Jessica Alba", "12345");
23  
24      assertTrue(!a.equals(b));
25      assertTrue(a.equals(c));
26      assertTrue(!b.equals(c));
27    }

29    @Test
30    public void testHashCode() {
31      Actor a = new Actor("Jessica Biel", "12345");
32      Actor b = new Actor("Jessica Biel", "67890");
33      Actor c = new Actor("Jessica Alba", "12345");
34  
35      assertTrue(a.hashCode() != b.hashCode());
36      assertTrue(a.hashCode() == c.hashCode());
37      assertTrue(b.hashCode() != c.hashCode());
38    }

40    @Test
41    public void testBasicGetters() {
42      Actor a = new Actor("Jessica Biel", "12345");
43      Actor b = new Actor("Jessica Alba", "67890");

45      assertTrue(a.getId().equals("12345"));
46      assertTrue(b.getId().equals("67890"));

51    @Test
52    public void testNullGetters() {
53      Actor a = new Actor("Jessica Biel", "12345");
54      Actor b = new Actor("Jessica Alba", "67890");
55  
56      assertTrue(a.getEdge(a.getId()) == -1);
57      assertTrue(a.getEdge(b.getId()) == -1);
58      assertTrue(b.getEdge(a.getId()) == -1);
59      assertTrue(b.getEdge(b.getId()) == -1);
60  
61      assertTrue(a.getEdgeName(a.getId()) == null);
62      assertTrue(a.getEdgeName(b.getId()) == null);
63      assertTrue(b.getEdgeName(a.getId()) == null);
64      assertTrue(b.getEdgeName(b.getId()) == null);
65    }

67    @Test
68    public void testNeighborFunctions() {
69      Actor a = new Actor("Jessica Biel", "12345");
70      Actor b = new Actor("Jessica Alba", "67890");
71      Actor c = new Actor("Sarah Jessica Parker", "13579");
72  
73      Film f = new Film("Movie", "24680", b, .5);
74      Film g = new Film("Film", "54321", c, .3);
75  
76      a.addNeighbor(f);
77      a.addNeighbor(g);
78  
79      assertTrue(a.getEdge("67890") == .5);
80      assertTrue(a.getEdge("13579") == .3);
81      assertTrue(b.getEdge("12345") == -1);
82      assertTrue(c.getEdge("12345") == -1);
83  
84      assertTrue(a.getEdgeName("67890").equals("Movie"));
85      assertTrue(a.getEdgeName("13579").equals("Film"));
86    }
