20    public void oneWordTrie() {
21      List<String> s = new ArrayList<>();
22      s.add("Sylvester Stallone");
23      Trie t = new Trie(s);
24  
25      assertTrue(t.isWord("Sylvester Stallone"));
26      assertTrue(t.isWord("sylvester stallone"));
27      assertTrue(t.isWord("SYLVESTER stallone"));
28      assertTrue(!t.isWord("Sylvester"));
29      assertTrue(!t.isWord("Stallone"));
30      assertTrue(!t.isWord("Sylvester "));
31      assertTrue(!t.isWord(" Stallone"));
32    }

34    @Test
35    public void severalWordsSamePrefix() {
36      List<String> s = new ArrayList<>();
37      s.add("A.J. Damato");
38      s.add("A.J. Drago");
39      s.add("A.J. Draven");
40      s.add("A.J. Khan");
41      s.add("A.S. Gyani");
42      Trie t = new Trie(s);
43      List<String> p = t.prefixComplete("a.");
44      List<String> q = t.prefixComplete("A.J.");
45      List<String> r = t.prefixComplete("a.j. d");
46  
47      assertTrue(p.contains("A.J. Damato"));
48      assertTrue(p.contains("A.J. Drago"));
49      assertTrue(p.contains("A.J. Draven"));
50      assertTrue(p.contains("A.J. Khan"));
51      assertTrue(p.contains("A.S. Gyani"));
52      assertTrue(p.size() == 5);
53  
54      assertTrue(q.contains("A.J. Damato"));
55      assertTrue(q.contains("A.J. Drago"));
56      assertTrue(q.contains("A.J. Draven"));
57      assertTrue(q.contains("A.J. Khan"));
58      assertTrue(!q.contains("A.S. Gyani"));
59      assertTrue(q.size() == 4);
60  
61      assertTrue(r.contains("A.J. Damato"));
62      assertTrue(r.contains("A.J. Drago"));
63      assertTrue(r.contains("A.J. Draven"));
64      assertTrue(!r.contains("A.J. Khan"));
65      assertTrue(!r.contains("A.S. Gyani"));
66      assertTrue(r.size() == 3);
67  
68      assertTrue(t.isWord("A.J. Damato"));
69      assertTrue(t.isWord("a.j. drago"));
70      assertTrue(t.isWord("A.j. DrAVen"));
71      assertTrue(t.isWord("A.J. KHAN"));
72      assertTrue(t.isWord("a.s. gyani"));
73    }

75    @Test
76    public void testLED() {
77      List<String> s = new ArrayList<>();
78      s.add("Aaron Starr");
79      s.add("Aaron Stark");
80      s.add("Aaron Stalls");
81      s.add("Erin Starr");
82      s.add("Erik Stark");
83      Trie t = new Trie(s);
84      List<String> l = t.levenshteinED("Aaron Star", 1);
85      List<String> m = t.levenshteinED("Aaron Sta", 3);
86      List<String> n = t.levenshteinED("Aaron Starr", 3);
87      List<String> o = t.levenshteinED("Erik Stark", 0);
88  
89      assertTrue(l.size() == 2);
90      assertTrue(l.contains("Aaron Starr"));
91      assertTrue(l.contains("Aaron Stark"));
92  
93      assertTrue(m.size() == 3);
94      assertTrue(m.contains("Aaron Starr"));
95      assertTrue(m.contains("Aaron Stark"));
96      assertTrue(m.contains("Aaron Stalls"));
97  
98      assertTrue(n.size() == 4);
99      assertTrue(n.contains("Aaron Starr"));
100     assertTrue(n.contains("Aaron Stark"));
101     assertTrue(n.contains("Aaron Stalls"));
102     assertTrue(n.contains("Erin Starr"));
103 
104     assertTrue(o.size() == 1);
105     assertTrue(o.contains("Erik Stark"));
